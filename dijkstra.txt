#include <bits/stdc++.h>
using namespace std;

int main() {
	vector<vector<pair<int,int> > > v(6);
	pair<int,int> p;
	int m,x,y,w;
	cin>>m;
	while(m--){
		cin>>x>>y>>w;
		v[x].push_back(make_pair(y,w));
		v[y].push_back(make_pair(x,w));
	}
	vector<int> d(6,99999);
   	set<pair<int,int> > s;
	s.insert(make_pair(0,0));
	d[0]=0;
	while(!s.empty()){
		p=*(s.begin());
		s.erase(s.begin());
		int dist=p.first,node=p.second;
		for(auto it=v[node].begin();it!=v[node].end();++it){
			p=(*it);
			int v2=p.first,weight=p.second;
              if(d[v2]>(dist+weight)){
              if(d[v2]!=99999)	s.erase(s.find(make_pair(d[v2],v2)));
              	s.insert(make_pair(dist+weight,v2));
              	d[v2]=dist+weight;
              }
		}
	}
	for(int i=0;i<6;++i)cout<<d[i]<<endl;
	return 0;
}
